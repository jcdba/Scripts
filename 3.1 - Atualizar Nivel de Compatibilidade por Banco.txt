--ATUALIZAR NIVEL DE COMPATIBILIDADE POR BANCO

--VERIFICANDO VERSÃO SQL
SELECT SERVERPROPERTY('ProductVersion')

--LOCALIZANDO VERSÃO DE COMPATIBILIDADE DO BANCO
SELECT name, compatibility_level FROM sys.databases;

--APLICANDO ATUALIZAÇÃO NIVEL DE COMPATIBILIDADE POR BANCO
ALTER DATABASE #NomeDatabase  --INFORMAR NOME DO BANCO 
SET COMPATIBILITY_LEVEL = 110;  --INFORMAR GRAU DE COMPATIBILIDADE CONFORME VERSÃO DO SQLSERVER
GO
